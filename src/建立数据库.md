CREATE TABLE RdpgCode.dbo.CodeDatabase (
    Id INT PRIMARY KEY IDENTITY(1,1),    -- 主键ID
    -- 报告信息    
   ProjectName NVARCHAR(200) NOT NULL,   -- 项目名称
   EvaluationAmount DECIMAL(18,2),        -- 评估金额
   ReportTime DATE,                    -- 报告时间
   ReportNumber NVARCHAR(100),              -- 报告号

   -- 签字人员A信息
    SignerA_Name NVARCHAR(50),               -- 签字人员A
    SignerA_Number NVARCHAR(50),              -- 签字人员A编号  
    -- 签字人员B信息
    SignerB_Name NVARCHAR(50),               -- 签字人员B
    SignerB_Number NVARCHAR(50),              -- 签字人员B编号
);

--引入“映射混淆码”
CREATE TABLE RdpgCode.dbo.CqrdpgCodepageDecodeMapping (
    Id INT IDENTITY(1,1) PRIMARY KEY,  -- 自增主键
    OriginalValue INT NOT NULL,             -- 需要解码的数据 (0-9)
    DecodedText NVARCHAR(255) NOT NULL, -- 解码后的文本
    Description NVARCHAR(500),          -- 可选：描述信息
);
